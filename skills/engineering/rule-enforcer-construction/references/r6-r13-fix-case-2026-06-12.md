**R9（从config.yaml读取实际provider数量）**

```python
def check_dual_ai():
    """检查是否有至少2个provider实现双AI互审"""
    config_path = os.path.expanduser("~/.hermes/config.yaml")
    if not os.path.exists(config_path):
        return "无法检查双AI: config.yaml不存在"
    
    import yaml
    with open(config_path) as f:
        config = yaml.safe_load(f)
    
    providers = config.get("providers", {})
    # 排除only_global/only_available/null/disabled
    active_providers = [
        k for k, v in providers.items()
        if isinstance(v, dict)
        and v.get("api_key") is not None
        and v.get("api_key", "").startswith("sk-")
    ]
    
    if len(active_providers) >= 2:
        return f"通过: {len(active_providers)}个活跃provider"
    else:
        return f"警告: 仅{len(active_providers)}个活跃provider，无法双AI互审"
```
